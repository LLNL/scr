33 /*
34  * The scr_transfer program is a deamon process that SCR launches as
35  * one process per compute node. It sleeps in the background, waking
36  * periodically to read the transfer.scrinfo file from cache, which
37  * the library will fill with info regarding asynchronous flushes.
38  * When it detects a START flag in this file, it slowly copies files
39  * from cache to the parallel file system.
40 */

77 int close_files(char* src, int* fd_src, char* dst, int* fd_dst)
78 {
79  /* close the source file, if it's open */
80  if (*fd_src >= 0) {
81  scr_close(src, *fd_src);
82  *fd_src = -1;
83  }
84 
85  /* close the destination file, if it's open */
86  if (*fd_dst >= 0) {
87  scr_close(dst, *fd_dst);
88  *fd_dst = -1;
89  }
90 
91  return SCR_SUCCESS;

121 
122  /* set position to 0 */
123  if (position != NULL) {
124  *position = 0;

130 /* given a hash of files and a file name, check whether the named
131  * file needs data transfered, if so, strdup its destination name
132  * and set its position and filesize */
133 int need_transfer(scr_hash* files, char* src, char** dst, off_t* position, off_t* filesize)
134 {
135  /* check that we got a hash of files and a file name */
136  if (files == NULL || src == NULL) {
137  return SCR_FAILURE;
138  }
139 
140  /* lookup the specified file in the hash */
141  scr_hash* file_hash = scr_hash_get(files, src);
142  if (file_hash == NULL) {
143  return SCR_FAILURE;
144  }
145 
146  /* extract the values for file size, bytes written, and destination */
147  unsigned long size, written;
148  char* dest;
149  if (scr_hash_util_get_bytecount(file_hash, SCR_TRANSFER_KEY_SIZE, &size) == SCR_SUCCESS &&
150  scr_hash_util_get_bytecount(file_hash, SCR_TRANSFER_KEY_WRITTEN, &written) == SCR_SUCCESS &&
151  scr_hash_util_get_str(file_hash, SCR_TRANSFER_KEY_DESTINATION, &dest) == SCR_SUCCESS)
152  {
153  /* if the bytes written value is less than the file size,
154  * we've got a valid file */
155  if (written < size) {
156  /* got our file, fill in output parameters */
157  *dst = strdup(dest);
158  /* TODO: check for error */
159  *position = (off_t) written;
160  *filesize = (off_t) size;
161  return SCR_SUCCESS;
162  }
163  }
164 
165  return SCR_FAILURE;
166 }

168 /* given a hash of transfer file data, look for a file which needs to
169  * be transfered. If src file is set, try to continue with that file,
170  * otherwise, pick the first available file */
171 int find_file(scr_hash* hash, char** src, char** dst, off_t* position, off_t* filesize)
172 {
173  int found_a_file = 0;
174 
175  scr_hash* files = scr_hash_get(hash, SCR_TRANSFER_KEY_FILES);
176  if (files != NULL) {
177  /* if we're given a file name, try to continue with that file */
178  if (!found_a_file && src != NULL && *src != NULL) {
179  /* src was set, so assume dst is also set, create a dummy dst
180  * variable to hold the string which may be strdup'd in
181  * need_transfer call */
182  char* tmp_dst = NULL;
183  if (need_transfer(files, *src, &tmp_dst, position, filesize) == SCR_SUCCESS) {
184  /* can continue with the same file (position may have been
185  * updated though) */
186  found_a_file = 1;
187 
188  /* free the dummy */
189  /* TODO: note that if destination has been updated, we're
190  * ignoring that change */
191  free(tmp_dst);
192  } else {
193  /* otherwise, this file no longer needs transfered,
194  * so free the strings */
195  clear_parameters(src, dst, position);
196  }
197  }
198 
199  /* if we still don't have a file, scan the hash and use the first
200  * file we find */
201  if (!found_a_file) {
202  scr_hash_elem* elem;
203  for (elem = scr_hash_elem_first(files);
204  elem != NULL;
205  elem = scr_hash_elem_next(elem))
206  {
207  /* get the filename */
208  char* name = scr_hash_elem_key(elem);
209 
210  /* check whether this file needs transfered */
211  if (name != NULL &&
212  need_transfer(files, name, dst, position, filesize) == SCR_SUCCESS)
213  {
214  /* found a file, copy its name (the destination and postion
215  * are set in need_transfer) */
216  *src = strdup(name);
217  found_a_file = 1;
218  break;
219  }
220  }
221  }
222  }
223 
224  /* if we didn't find a file, set src and dst to NULL and set
225  * position to 0 */
226  if (!found_a_file) {
227  clear_parameters(src, dst, position);
228  return SCR_FAILURE;
229  }
230 
231  return SCR_SUCCESS;
232 }

234 /* writes the specified command to the transfer file */
235 int set_transfer_file_state(char* s, int done)
236 {
237  /* get a hash to store file data */
238  scr_hash* hash = scr_hash_new();
239 
240  /* attempt to read the file transfer file */
241  int fd = -1;
242  if (scr_hash_lock_open_read(scr_transfer_file, &fd, hash) == SCR_SUCCESS) {
243  /* set the state */
244  scr_hash_util_set_str(hash, SCR_TRANSFER_KEY_STATE, s);
245 
246  /* set the flag if we're done */
247  if (done) {
248  scr_hash_set_kv(hash, SCR_TRANSFER_KEY_FLAG, SCR_TRANSFER_KEY_FLAG_DONE);
249  }
250 
251  /* write the hash back to the file */
252  scr_hash_write_close_unlock(scr_transfer_file, &fd, hash);
253  }
254 
255  /* delete the hash */
256  scr_hash_delete(&hash);
257 
258  return SCR_SUCCESS;
259 }

261 /* read the transfer file and set our global variables to match */
262 scr_hash* read_transfer_file()
263 {
264  char* value = NULL;
265 
266  /* get a new hash to store the file data */
267  scr_hash* hash = scr_hash_new();
268 
269  /* open transfer file with lock */
270  scr_hash_read_with_lock(scr_transfer_file, hash);
271 
272  /* read in our allowed bandwidth value */
273  value = scr_hash_elem_get_first_val(hash, SCR_TRANSFER_KEY_BW);
274  if (value != NULL) {
275  double bw;
276  if (scr_atod(value, &bw) == SCR_SUCCESS) {
277  /* got a new bandwidth value, set our global variable */
278  bytes_per_second = bw;
279  } else {
280  /* could not interpret bandwidth value */
281  scr_err("scr_transfer: Ignoring invalid BW value in %s @ %s:%d",
282  scr_transfer_file, __FILE__, __LINE__
283  );
284  }
285  } else {
286  /* couldn't find a BW field, so disable this limit */
287  bytes_per_second = 0.0;
288  }
289 
290  /* read in our allowed percentage of runtime value */
291  value = scr_hash_elem_get_first_val(hash, SCR_TRANSFER_KEY_PERCENT);
292  if (value != NULL) {
293  double percent;
294  if (scr_atod(value, &percent) == SCR_SUCCESS) {
295  /* got a new bandwidth value, set our global variable */
296  percent_runtime = percent / 100.0;
297  } else {
298  /* could not interpret bandwidth value */
299  scr_err("scr_transfer: Ignoring invalid PERCENT value in %s @ %s:%d",
300  scr_transfer_file, __FILE__, __LINE__
301  );
302  }
303  } else {
304  /* couldn't find a PERCENT field, so disable this limit */
305  percent_runtime = 0.0;
306  }
307 
308  /* check for DONE flag */
309  int done = 0;
310  scr_hash* done_hash = scr_hash_get_kv(hash, SCR_TRANSFER_KEY_FLAG, SCR_TRANSFER_KEY_FLAG_DONE);
311  if (done_hash != NULL) {
312  done = 1;
313  }
314 
315  /* check for latest command */
316  state = STOPPED;
317  value = scr_hash_elem_get_first_val(hash, SCR_TRANSFER_KEY_COMMAND);
318  if (value != NULL) {
319  if (strcmp(value, SCR_TRANSFER_KEY_COMMAND_EXIT) == 0) {
320  /* close files and exit */
321  scr_dbg(0,"scr_transfer instructed to exit @%s:%d\n",
322  __FILE__, __LINE__
323  );
324  set_transfer_file_state(SCR_TRANSFER_KEY_STATE_EXIT, 0);
325  keep_running = 0;
326  } else if (strcmp(value, SCR_TRANSFER_KEY_COMMAND_STOP) == 0) {
327  /* just stop, nothing else to do here */
328  scr_dbg(0,"scr_transfer instructed to STOP @%s:%d\n",
329  __FILE__, __LINE__
330  );
331  } else if (strcmp(value, SCR_TRANSFER_KEY_COMMAND_RUN) == 0) {
332  /* found the RUN command, if the DONE flag is not set,
333  * set our state to running and update the transfer file */
334  scr_dbg(0,"scr_transfer instructed to RUN @%s:%d\n",
335  __FILE__, __LINE__
336  );
337  if (!done) {
338  state = RUNNING;
339  set_transfer_file_state(SCR_TRANSFER_KEY_STATE_RUN, 0);
340  }
341  } else {
342  scr_err("scr_transfer: Unknown command %s in %s @ %s:%d",
343  value, scr_transfer_file, __FILE__, __LINE__
344  );
345  }
346  }
347 
348  /* ensure that our current state is always recorded in the file
349  * (the file may have been deleted since we last wrote our state
350  * to it) */
351  value = scr_hash_elem_get_first_val(hash, SCR_TRANSFER_KEY_STATE);
352  if (value == NULL) {
353  if (state == STOPPED) {
354  set_transfer_file_state(SCR_TRANSFER_KEY_STATE_STOP, 0);
355  } else if (state == RUNNING) {
356  set_transfer_file_state(SCR_TRANSFER_KEY_STATE_RUN, 0);
357  } else {
358  scr_err("scr_transfer: Unknown state %d @ %s:%d",
359  state, __FILE__, __LINE__
360  );
361  }
362  }
363 
364  return hash;
365 }

367 /* given src and dst, find the matching entry in the transfer file and
368  * update the WRITTEN field to the given position in that file */
369 int update_transfer_file(char* src, char* dst, off_t position)
370 {
371  /* create a hash to store data from file */
372  scr_hash* hash = scr_hash_new();
373 
374  /* attempt to open transfer file with lock,
375  * if we fail to open the file, don't bother writing to it */
376  int fd = -1;
377  if (scr_hash_lock_open_read(scr_transfer_file, &fd, hash) == SCR_SUCCESS) {
378  /* search for the source file, and update the bytes written if found */
379  if (src != NULL) {
380  scr_hash* file_hash = scr_hash_get_kv(hash, SCR_TRANSFER_KEY_FILES, src);
381  if (file_hash != NULL) {
382  /* update the bytes written field */
383  scr_hash_util_set_bytecount(file_hash, "WRITTEN", (uint64_t) position);
384  }
385  }
386 
387  /* close the transfer file and release the lock */
388  scr_hash_write_close_unlock(scr_transfer_file, &fd, hash);
389  }
390 
391  /* free the hash */
392  scr_hash_delete(&hash);
393 
394  return SCR_SUCCESS;
395 }

397 /* returns 1 if file1 and file2 are different
398  * (NULL pointers are allowed) */
399 int bool_diff_files(char* file1, char* file2)
400 {
401  /* if file2 is set but file1 is not, they're different */
402  if (file1 != NULL && file2 == NULL) {
403  return 1;
404  }
405 
406  /* if file1 is set but file2 is not, they're different */
407  if (file1 == NULL && file2 != NULL) {
408  return 1;
409  }
410 
411  /* if file1 and file2 and both set,
412  * check whether the strings are the same */
413  if (file1 != NULL && file2 != NULL && strcmp(file1, file2) != 0) {
414  return 1;
415  }
416 
417  /* otherwise, consider file1 to be the same as file2 */
418  return 0;
419 }

440 
441  return SCR_SUCCESS;

444 int main (int argc, char *argv[])
445 {
446 
447 #ifdef HAVE_LIBCPPR
448  /* this program is a noop if we have libcppr, as libcppr
449  * will do the file transfers asynchronously for us */
450  scr_dbg(1,"scr_transfer shutting down since libcppr is present\n");
451 #else
452  /* check that we were given at least one argument
453  * (the transfer file name) */
454  if (argc != 2) {
455  printf("Usage: scr_transfer <transferfile>\n");
456  return 1;
457  }
458 
459  /* record the name of the transfer file */
460  scr_transfer_file = strdup(argv[1]);
461  if (scr_transfer_file == NULL) {
462  scr_err("scr_transfer: Copying transfer file name @ %s:%d",
463  __FILE__, __LINE__
464  );
465  return 1;
466  }
467 
468  /* initialize our tracking variables */
469  read_params();
470 
471  /* get file io mode */
472  mode_t mode_file = scr_getmode(1, 1, 0);
473 
474  /* build the control directory name: CNTL_BASE/username/scr.jobid */
475  scr_path* scr_cntl_path = scr_path_from_str(argv[1]);
476  scr_path_dirname(scr_cntl_path);
477  char* path = scr_path_strdup(scr_cntl_path);
478 
479  /* create the control directory */
480  if (scr_mkdir(path, S_IRWXU | S_IRWXG) != SCR_SUCCESS) {
481  scr_abort(-1, "Failed to create control directory: %s @ %s:%d",
482  path, __FILE__, __LINE__
483  );
484  }
485 
486  scr_path_delete(&scr_cntl_path);
487  scr_free(&path);
488 
489  /* we cache the opened file descriptors to avoid extra opens,
490  * seeks, and closes */
491  int fd_src = -1;
492  int fd_dst = -1;
493 
494  char* new_file_src = NULL;
495  char* old_file_src = NULL;
496  char* new_file_dst = NULL;
497  char* old_file_dst = NULL;
498 
499  off_t new_position = 0;
500  off_t old_position = 0;
501 
502  /* start in the stopped state */
503  state = STOPPED;
504  set_transfer_file_state(SCR_TRANSFER_KEY_STATE_STOP, 0);
505 
506  /* TODO: enable this value to be set from config file */
507  /* TODO: page-align this buffer for faster performance */
508  /* allocate our file copy buffer */
509  size_t bufsize = scr_file_buf_size;
510  char* buf = malloc(bufsize);
511  if (buf == NULL) {
512  scr_err("scr_transfer: Failed to allocate %llu bytes for file copy buffer @ %s:%d",
513  (unsigned long long) bufsize, __FILE__, __LINE__
514  );
515  return 1;
516  }
517 
518  int nread = 0;
519  double secs_run = 0.0;
520  double secs_slept = 0.0;
521  double secs_run_start = scr_seconds();
522  double secs_run_end = secs_run_start;
523  double secs_last_write = secs_run_start;
524  scr_hash* hash = scr_hash_new();
525  while (keep_running) {
526  /* loop here sleeping and checking transfer file periodically
527  * until state changes and / or some time elapses */
528  /* reset our timer for our last write */
529  double secs_remain = scr_transfer_secs;
530  while (keep_running && (state == STOPPED || secs_remain > 0.0)) {
531  /* remember our current state before reading transfer file */
532  int old_state = state;
533 
534  /* read the transfer file, which fills in our hash and
535  * also updates state and bytes_per_second */
536  scr_hash_delete(&hash);
537  hash = read_transfer_file();
538 
539  /* compute time we should sleep before writing more data based
540  * on bandwidth and percent of runtime limits */
541  if (state == RUNNING) {
542  /* get the current time */
543  double secs_now = scr_seconds();
544 
545  /* based on the amount we last wrote and our allocated bandwidth,
546  * compute time we need to sleep before attempting our next write */
547  double secs_remain_bw = 0.0;
548  if (nread > 0 && bytes_per_second > 0.0) {
549  double secs_to_wait_bw = (double) nread / bytes_per_second;
550  double secs_waited_bw = secs_now - secs_last_write;
551  secs_remain_bw = secs_to_wait_bw - secs_waited_bw;
552  }
553 
554  /* based on the percentage of time we are allowed to be running,
555  * compute time we need to sleep before attempting our next write */
556  double secs_remain_runtime = 0.0;
557  if (percent_runtime > 0.0) {
558  /* stop the run clock, add to the run time,
559  * and restart the run clock */
560  secs_run_end = secs_now;
561  secs_run += secs_run_end - secs_run_start;
562  secs_run_start = secs_run_end;
563 
564  /* compute our total time, and the time we need to sleep */
565  double secs_total = secs_run + secs_slept;
566  secs_remain_runtime = secs_run / percent_runtime - secs_total;
567  }
568 
569  /* take the maximum of these two values */
570  secs_remain = secs_remain_bw;
571  if (secs_remain_runtime > secs_remain) {
572  secs_remain = secs_remain_runtime;
573  }
574  }
575 
576  /* check for a state transition */
577  if (state != old_state) {
578  if (state == RUNNING) {
579  /* if we switched to RUNNING, kick out without sleeping and
580  * reset the total run and sleep times */
581  secs_remain = 0.0;
582  secs_run = 0.0;
583  secs_slept = 0.0;
584  } else if (state == STOPPED) {
585  /* if we switched to STOPPED, close our files if open */
586  close_files(new_file_src, &fd_src, new_file_dst, &fd_dst);
587  clear_parameters(&new_file_src, &new_file_dst, &new_position);
588  clear_parameters(&old_file_src, &old_file_dst, &old_position);
589 
590  /* after closing our files, update our state in the transfer file */
591  set_transfer_file_state(SCR_TRANSFER_KEY_STATE_STOP, 0);
592  }
593  }
594 
595  /* assume we can sleep for the full remainder of the time */
596  double secs = secs_remain;
597 
598  /* if we're not running, always sleep for the full time */
599  if (state != RUNNING) {
600  secs = scr_transfer_secs;
601  }
602 
603  /* set a maximum time to sleep before we read the hash file again
604  * (ensures some responsiveness) */
605  if (secs > scr_transfer_secs) {
606  secs = scr_transfer_secs;
607  }
608 
609  /* sleep if we need to */
610  if (secs > 0.0) {
611  /* stop the run clock and add to the total run time */
612  secs_run_end = scr_seconds();
613  secs_run += secs_run_end - secs_run_start;
614 
615  /* sleep */
616  usleep((unsigned long) (secs * 1000000.0));
617  secs_slept += secs;
618  secs_remain -= secs;
619 
620  /* restart the run clock */
621  secs_run_start = scr_seconds();
622  }
623  }
624 
625  /* write data out */
626  if (state == RUNNING) {
627  /* look for a new file to transfer */
628  off_t filesize = 0;
629  find_file(hash, &new_file_src, &new_file_dst, &new_position, &filesize);
630 
631  /* if we got a new file, close the old one (if open),
632  * open the new file */
633  if (bool_diff_files(new_file_src, old_file_src)) {
634  /* close the old descriptor if it's open */
635  if (fd_src >= 0) {
636  scr_close(old_file_src, fd_src);
637  fd_src = -1;
638  }
639 
640  /* delete the old file name if we have one */
641  if (old_file_src != NULL) {
642  free(old_file_src);
643  old_file_src = NULL;
644  }
645 
646  /* reset our position counter */
647  old_position = 0;
648 
649  /* open the file and remember the filename if we have one */
650  if (new_file_src != NULL) {
651  fd_src = scr_open(new_file_src, O_RDONLY);
652  /* TODO: check for errors here */
