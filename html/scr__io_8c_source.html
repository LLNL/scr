91 /* returns user's current mode as determine by his umask */
92 mode_t scr_getmode(int read, int write, int execute)
93 {
94  /* lookup current mask and set it back */
95  mode_t old_mask = umask(S_IWGRP | S_IWOTH);
96  umask(old_mask);
97 
98  mode_t bits = 0;
99  if (read) {
100  bits |= (S_IRUSR | S_IRGRP | S_IROTH);
101  }
102  if (write) {
103  bits |= (S_IWUSR | S_IWGRP | S_IWOTH);
104  }
105  if (execute) {
106  bits |= (S_IXUSR | S_IXGRP | S_IXOTH);
107  }
108 
109  /* convert mask to mode */
110  mode_t mode = bits & ~old_mask & 0777;
111  return mode;
112 }

114 /* open file with specified flags and mode, retry open a few times on failure */
115 int scr_open(const char* file, int flags, ...)
116 {
117  /* extract the mode (see man 2 open) */
118  int mode_set = 0;
119  mode_t mode = 0;
120  if (flags & O_CREAT) {
121  va_list ap;
122  va_start(ap, flags);
123  mode = va_arg(ap, mode_t);
124  va_end(ap);
125  mode_set = 1;
126  }
127 
128  int fd = -1;
129  if (mode_set) {
130  fd = open(file, flags, mode);
131  } else {
132  fd = open(file, flags);
133  }
134  if (fd < 0) {
135  scr_dbg(1, "Opening file: open(%s) errno=%d %s @ %s:%d",
136  file, errno, strerror(errno), __FILE__, __LINE__
137  );
138 
139  /* try again */
140  int tries = SCR_OPEN_TRIES;
141  while (tries && fd < 0) {
142  usleep(SCR_OPEN_USLEEP);
143  if (mode_set) {
144  fd = open(file, flags, mode);
145  } else {
146  fd = open(file, flags);
147  }
148  tries--;
149  }
150 
151  /* if we still don't have a valid file, consider it an error */
152  if (fd < 0) {
153  scr_err("Opening file: open(%s) errno=%d %s @ %s:%d",
154  file, errno, strerror(errno), __FILE__, __LINE__
155  );
156  }
157  }
158  return fd;
159 }

161 /* fsync and close file */
162 int scr_close(const char* file, int fd)
163 {
164  /* fsync first */
165  if (fsync(fd) < 0) {
166  /* print warning that fsync failed */
167  scr_dbg(2, "Failed to fsync file descriptor: %s errno=%d %s @ %s:%d",
168  file, errno, strerror(errno), __FILE__, __LINE__
169  );
170  }
171 
172  /* now close the file */
173  if (close(fd) != 0) {
174  /* hit an error, print message */
175  scr_err("Closing file descriptor %d for file %s: errno=%d %s @ %s:%d",
176  fd, file, errno, strerror(errno), __FILE__, __LINE__
177  );
178  return SCR_FAILURE;
179  }
180 
181  return SCR_SUCCESS;
182 }

271 /* opens specified file and waits on a lock before returning the file descriptor */
272 int scr_open_with_lock(const char* file, int flags, mode_t mode)
273 {
274  /* open the file */
275  int fd = scr_open(file, flags, mode);
276  if (fd < 0) {
277  scr_err("Opening file for write: scr_open(%s) errno=%d %s @ %s:%d",
278  file, errno, strerror(errno), __FILE__, __LINE__
279  );
280  return fd;
281  }
282 
283  /* acquire an exclusive file lock */
284  int ret = scr_file_lock_write(file, fd);
285  if (ret != SCR_SUCCESS) {
286  close(fd);
287  return ret;
288  }
289 
290  /* return the opened file descriptor */
291  return fd;
292 }

320 /* reliable read from file descriptor (retries, if necessary, until hard error) */
321 ssize_t scr_read(const char* file, int fd, void* buf, size_t size)
322 {
323  ssize_t n = 0;
324  int retries = 10;
325  while (n < size)
326  {
327  int rc = read(fd, (char*) buf + n, size - n);
328  if (rc > 0) {
329  n += rc;
330  } else if (rc == 0) {
331  /* EOF */
332  return n;
333  } else { /* (rc < 0) */
334  /* got an error, check whether it was serious */
335  if (errno == EINTR || errno == EAGAIN) {
336  continue;
337  }
338 
339  /* something worth printing an error about */
340  retries--;
341  if (retries) {
342  /* print an error and try again */
343  scr_err("Error reading %s: read(%d, %x, %ld) errno=%d %s @ %s:%d",
344  file, fd, (char*) buf + n, size - n, errno, strerror(errno), __FILE__, __LINE__
345  );
346  } else {
347  /* too many failed retries, give up */
348  scr_err("Giving up read of %s: read(%d, %x, %ld) errno=%d %s @ %s:%d",
349  file, fd, (char*) buf + n, size - n, errno, strerror(errno), __FILE__, __LINE__
350  );
351  exit(1);
352  }
353  }
354  }
355  return n;
356 }

478 /* read line from file into buf with given size */
479 ssize_t scr_read_line(const char* file, int fd, char* buf, size_t size)
480 {
481  /* read up to size-1 bytes from fd into buf until we find a newline or EOF */
482  ssize_t n = 0;
483  int found_end = 0;
484  while (n < size-1 && !found_end) {
485  /* read a character from the file */
486  char c;
487  ssize_t nread = scr_read(file, fd, &c, sizeof(c));
488 
489  if (nread > 0) {
490  /* we read a character, copy it over to the buffer */
491  buf[n] = c;
492  n++;
493 
494  /* check whether we hit the end of the line */
495  if (c == '\n') {
496  found_end = 1;
497  }
498  } else if (nread == 0) {
499  /* we hit the end of the file */
500  found_end = 1;
501  } else { /* nread < 0 */
502  /* we hit an error */
503  scr_err("Error reading from file %s @ %s:%d",
504  file, __FILE__, __LINE__
505  );
506  return -1;
507  }
508  }
509 
510  /* tack on the NULL character */
511  buf[n] = '\0';
512 
513  /* if we exit the while loop but didn't find the end of the line, the buffer was too small */
514  if (!found_end) {
515  scr_err("Buffer too small to read line from file %s @ %s:%d",
516  file, __FILE__, __LINE__
517  );
518  return -1;
519  }
520 
521  /* NOTE: we don't want to count the NULL which we added, but there is no need to adjust n here */
522  return n;
523 }

525 /* write a formatted string to specified file descriptor */
526 ssize_t scr_writef(const char* file, int fd, const char* format, ...)
527 {
528  /* write the formatted string to a buffer */
529  char buf[SCR_MAX_LINE];
530  va_list argp;
531  va_start(argp, format);
532  int n = vsnprintf(buf, sizeof(buf), format, argp);
533  va_end(argp);
534 
535  /* check that our buffer was large enough */
536  if (sizeof(buf) <= n) {
537  /* TODO: instead of throwing a fatal error, we could allocate a bigger buffer and try again */
538 
539  scr_err("Buffer too small to hold formatted string for file %s @ %s:%d",
540  file, __FILE__, __LINE__
541  );
542  exit(1);
543  }
544 
545  /* write the string out to the file descriptor */
546  ssize_t rc = scr_write(file, fd, buf, n);
547 
548  return rc;
549 }

551 /* logically concatenate n opened files and read count bytes from this logical file into buf starting
552  * from offset, pad with zero on end if missing data */
553 int scr_read_pad_n(int n, char** files, int* fds,
554  char* buf, unsigned long count, unsigned long offset, unsigned long* filesizes)
555 {
556  int i = 0;
557  size_t pos = 0;
558  off_t nseek = 0;
559  off_t nread = 0;
560 
561  /* pass through files until we find the one containing our offset */
562  while (i < n && (nseek + filesizes[i]) <= offset) {
563  nseek += filesizes[i];
564  i++;
565  }
566 
567  /* seek to the proper position in the current file */
568  if (i < n) {
569  pos = offset - nseek;
570  nseek += pos;
571  if (lseek(fds[i], pos, SEEK_SET) == (off_t)-1) {
572  /* our seek failed, return an error */
573  scr_err("Failed to seek to byte %lu in %s @ %s:%d",
574  pos, files[i], __FILE__, __LINE__
575  );
576  return SCR_FAILURE;
577  }
578  }
579 
580  /* read data from files */
581  while (nread < count && i < n) {
582  /* assume we'll read the remainder of the current file */
583  size_t num_to_read = filesizes[i] - pos;
584 
585  /* if we don't need to read the whole remainder of the file, adjust to the smaller amount */
586  if (num_to_read > count - nread) {
587  num_to_read = count - nread;
588  }
589 
590  /* read data from file and add to the total read count */
591  if (scr_read_attempt(files[i], fds[i], buf + nread, num_to_read) != num_to_read) {
592  /* our read failed, return an error */
593  return SCR_FAILURE;
594  }
595  nread += num_to_read;
596 
597  /* advance to next file and seek to byte 0 */
598  i++;
599  if (i < n) {
600  pos = 0;
601  if (lseek(fds[i], pos, SEEK_SET) == (off_t)-1) {
602  /* our seek failed, return an error */
603  scr_err("Failed to seek to byte %lu in %s @ %s:%d",
604  pos, files[i], __FILE__, __LINE__
605  );
606  return SCR_FAILURE;
607  }
608  }
609  }
610 
611  /* if count is bigger than all of our file data, pad with zeros on the end */
612  if (nread < count) {
613  memset(buf + nread, 0, count - nread);
614  }
615 
616  return SCR_SUCCESS;
617 }

763 int scr_crc32(const char* filename, uLong* crc)
764 {

766  if (crc == NULL) {

768  }

926 #ifdef HAVE_LIBCPPR

945  if (src_fd < 0) {

947  src_file, errno, strerror(errno), __FILE__, __LINE__

955  if (dst_fd < 0) {

957  dst_file, errno, strerror(errno), __FILE__, __LINE__

959  scr_close(src_file, src_fd);

961  }

1012

1015  copying = 0;
1016  }
1017 
1018  /* check for a read error, stop copying and return an error */
1019  if (nread < 0) {

1021  copying = 0;
1022  rc = SCR_FAILURE;
1023  }
1024  }

1027  scr_free(&buf);
1028

1031  rc = SCR_FAILURE;
1032  }
1033  if (scr_close(src_file, src_fd) != SCR_SUCCESS) {
1034  rc = SCR_FAILURE;
1035  }
1036 
1037  /* unlink the file if the copy failed */
1038  if (rc != SCR_SUCCESS) {
1039  unlink(dst_file);
1040  }
1041 #ifdef HAVE_LIBCPPR
1042  else {

1046  }

1048
